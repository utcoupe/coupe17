146 template<typename T, uint8_t N> uint8_t os48::MemoryPool<T, N>::reserveBlock()
147 {
148  OS48_KERNEL_SAFETY_BLOCK
149  {
150  if (m_next_free_block != EOB)
151  {
152  uint8_t block = m_next_free_block;
153  m_next_free_block = m_blocks[block];
154  m_blocks[block] = EOB;
155  --m_used_block_count;
156  return block;
157  }
158  }
159 
160  return EOB;
161 }
